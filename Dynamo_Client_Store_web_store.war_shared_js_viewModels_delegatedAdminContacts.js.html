/**
 * @fileoverview scheduled Order List
 */
define(
  //-------------------------------------------------------------------
  // PACKAGE NAME
  //-------------------------------------------------------------------
  'viewModels/delegatedAdminContacts',

  //-------------------------------------------------------------------
  // DEPENDENCIES
  //-------------------------------------------------------------------
  ['knockout', 'paginated', 'pubsub', 'ccRestClient', 'ccConstants',
   'CCi18n','spinner', 'ccStoreConfiguration', 'viewModels/dynamicPropertyMetaContainer'],

   function(ko, Paginated, pubsub, ccRestClient, CCConstants, CCi18n, spinner, CCStoreConfiguration, DynamicPropertyMetaContainer) {

    "use strict";

    //-------------------------------------------------------
    // View Model
    //-------------------------------------------------------

    /**
     * delegatedAdminContacts,
     *    used when handling delegatedAdminContacts collections.
     * 
     * @public
     * @class
     * @name delegatedAdminContacts
     * @extends ccPaginated
     * 
     * @property {observable<boolean>} display=true Display results?
     * @property {integer} itemsPerPage=20 The number of items per page.
     * @property {integer} blockSize=20 The number of items per block.
     * @property {CCStoreConfiguration} storeConfiguration An instance of the cc-store-configuration containing store-configuration data.
     */
    function delegatedAdminContacts() {
      var self = this;
      // call super constructor
      Paginated.call(this);
      self.display = ko.observable(true);
      //Not being used from 17.2 version of delegatedAdminContacts widget
      //moved to user.js
      self.isDelegatedAdmin = ko.observable(false);
      self.dynamicPropertyMetaInfo = DynamicPropertyMetaContainer.getInstance();

      // additional public variables
      self.type = "delegatedAdminContacts";
      
      this.roleValue = "";
      this.searchValue = "";
      this.orgValue = "";
      this.fetchById = true;
      this.fetchByRepositoryId = false;
      this.itemsPerPage = 0;
      this.blockSize = 0;
      this.sortProperty="firstName:asc";
      this.storeConfiguration = CCStoreConfiguration.getInstance();
	  /** Default options for creating a spinner */
	  this.spinnerOptions = {
	    parent: '#CC-delegatedAdminContactsList-contactListing-info',
	    posTop: '0',
	    posLeft: '50%'
	  };
    }

    var Temp = function(){};
    Temp.prototype = Paginated.prototype;
    delegatedAdminContacts.prototype = new Temp();

    // setup the inheritance chain
    delegatedAdminContacts.prototype.constructor = delegatedAdminContacts;
    delegatedAdminContacts.prototype.itemsPerRow = ko.observable(1);
    
    /**
     * Load orders starting from the specified starting index.
     * @function scheduledOrderList#fetchBlock
     * @param {integer} startingIndex the index of the first record to load.
     */
    delegatedAdminContacts.prototype.fetchBlock = function(startingIndex) {
      var url, inputData;
      if(this.clearOnLoad) {
        this.clearData();
      }
      spinner.create(this.spinnerOptions);
      inputData = {};
      inputData[CCConstants.OFFSET] = startingIndex;
      inputData[CCConstants.LIMIT] = this.blockSize * 2;
      inputData[CCConstants.SORTS] = this.sortProperty;
      if((this.searchValue && this.searchValue != "") || (this.roleValue && this.roleValue != "")){
        if(this.searchValue != "" &&  this.roleValue != ""){
          inputData[CCConstants.Q] = "firstName co \""+this.searchValue+"\" OR lastName co \""+this.searchValue+"\" OR email co \""+this.searchValue+"\" AND roles eq \""+this.roleValue+"\"";  
        }else if(this.searchValue != "" && this.roleValue == ""){
          inputData[CCConstants.Q] = "firstName co \""+this.searchValue+"\" OR lastName co \""+this.searchValue+"\" OR email co \""+this.searchValue+"\"";
        }else if(this.searchValue == "" && this.roleValue != ""){
          if(this.fetchByRepositoryId === true){
        		inputData[CCConstants.Q] = "roles.id eq \""+this.roleValue+"\"";
          }else if(this.fetchById === true) {
            inputData[CCConstants.Q] = "roles eq \""+this.roleValue+"\"";
          } else {
            inputData[CCConstants.Q] = "roles.function eq \""+this.roleValue+"\"";
          }
        }
      }
      else if(this.orgValue && this.orgValue!= ""){
    	  inputData[CCConstants.SECONDARY_ORG_COUNT] = true;
    	  inputData[CCConstants.Q] = "parentOrganization eq \""+this.orgValue+ "\" OR secondaryOrganizations co \""+ this.orgValue+ "\"";
      }
      //TODO: End Point Call to fetch List of Contacts
      var contextObj = {};
      contextObj[CCConstants.ENDPOINT_KEY] = "listMembers";
      var filterKey = this.storeConfiguration.getFilterToUse(contextObj);
      if (filterKey) {
        inputData[CCConstants.FILTER_KEY] = filterKey;
      }
      ccRestClient.request("listMembers",inputData,this.successFunc.bind(this),this.errorFunc.bind(this));
     };

    /**
     * 'Get delegated Admin Contacts' success callback function.
     * Adds the order data to the view model.
     * @private
     * @function scheduledOrderList#successFunc
     * @param {Object} result The scheduledOrderList object.
     */
     delegatedAdminContacts.prototype.successFunc = function(data) {
       if (data && data.items) {
         for(var item = 0; item < data.items.length; item++) {
              if(data.items[item].roles && data.items[item].roles.length > 0 && data.items[item].roles[0]["function"] == "admin") {
                data.items[item].roles = data.items[item].roles.reverse();
              }
              var roleLength = data.items[item].roles.length;
              var roleString = '';
              for(var role = 0; role < roleLength; role++) {
                //data.items[item].roles[role]["function"] = data.items[item].roles[role]["function"].charAt(0).toUpperCase() + data.items[item].roles[role]["function"].slice(1);
                if(data.items[item].roles[role]["function"] == "buyer") {
                  roleString = roleString + CCConstants.BUYER_TITLE + ', ';
                }
                else if(data.items[item].roles[role]["function"] == "admin") {
                  roleString = roleString + CCConstants.ADMIN_TITLE + ', ';
                }
              }
              roleString = roleString.substring(0, roleString.length - 2);
              data.items[item]["roleString"] = roleString;
            }
         this.addData(data.items, data.total < 0 ? 0 : data.total, data.offset);
       }
       spinner.destroyWithoutDelay(this.spinnerOptions.parent);
     };

    /**
     * 'Get scheduledOrderList' error callback function.
     * Publishes the SCHEDULE_ORDERS_LIST_FAILED event.
     * @private
     * @function scheduledOrderList#errorFunc
     * @param {Object} result The error object.
     */
    delegatedAdminContacts.prototype.errorFunc = function(result) {
      spinner.destroyWithoutDelay(this.spinnerOptions.parent);
      $.Topic(pubsub.topicNames.DELEGATED_CONTACTS_LIST_FAILED).publishWith(result);
    };
    delegatedAdminContacts.prototype.refinedFetch = function() {
      var self = this;
      self.clearOnLoad = true;
      self.targetPage = 1;
      self.fetchBlock(0);
      self.scrollToTop = true;
    };
    
    return delegatedAdminContacts;
  }
);
